function parseBody(req) {
  return new Promise((resolve, reject) => {
    let body = '';
    req.on('data', chunk => body += chunk);
    req.on('end', () => {
      try { resolve(JSON.parse(body)); }
      catch (e) { reject(e); }
    });
  });
}

function proxyToExternal(req, res, targetHost, targetPath, extraHeaders) {
  const reqOpts = {
    hostname: targetHost,
    port: 443,
    path: targetPath,
    method: req.method,
    headers: {
      "host": targetHost,
      "user-agent": "Mozilla/5.0 ArkhamLocal/1.0",
      "accept": req.headers["accept"] || "*/*",
      "accept-encoding": "gzip, deflate, br",
      "accept-language": req.headers["accept-language"] || "en-US,en;q=0.9",
      ...extraHeaders
    }
  };
  const proxy = https.request(reqOpts, (proxyRes) => {
    const respHeaders = {
      ...proxyRes.headers,
      "access-control-allow-origin": "*",
      "access-control-allow-methods": "GET, POST, OPTIONS",
      "access-control-allow-headers": "Content-Type, Authorization, X-Client-Id"
    };
    // 移除可能引起问题的头
    delete respHeaders["content-security-policy"];
    delete respHeaders["x-frame-options"];
    res.writeHead(proxyRes.statusCode, respHeaders);
    proxyRes.pipe(res);
  });
  proxy.on("error", (e) => {
    console.error(`API proxy error [${targetHost}${targetPath}]:`, e.message);
    res.writeHead(502);
    res.end("API proxy error: " + e.message);
  });
  if (req.method !== "GET" && req.method !== "HEAD") {
    req.pipe(proxy);
  } else {
    proxy.end();
  }
}
